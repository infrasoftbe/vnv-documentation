# Models

## Nodes

<BrowserOnly>
{() => {

  const { default : ReactJson } = require('react-json-view');

  return <React.Fragment>
  {
    Object.keys(NodeContructor).map(( nodeType ) => {
      if(!NodeContructor[nodeType])return (<></>);
      return <div>
        <details>
          <summary><b>{nodeType}</b> Model</summary>
          <ReactJson src={new NodeContructor[nodeType]({
            meta : MetadataContructor[nodeType] ? new MetadataContructor[nodeType]({}) : {}
          })}></ReactJson>
        </details>
      </div>
    })
  }
  </React.Fragment>

}}
</BrowserOnly>
